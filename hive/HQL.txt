Required Queries
1. SELECT All Fields
SELECT * FROM employee;
2. WHERE Condition
SELECT *
FROM employee
WHERE salary > 30000;
3. GROUP BY Department
SELECT department,
AVG(salary) AS avg_salary
FROM employee
GROUP BY department;
4. HAVING Clause
SELECT department,
AVG(salary) AS avg_salary
FROM employee
GROUP BY department
HAVING AVG(salary) > 40000;
5. ORDER BY Employee Number
SELECT *
FROM employee
ORDER BY emp_id;
